-- LocalScript
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local keyCheckEvent = ReplicatedStorage:WaitForChild("KeyCheckEvent")

-- Menunggu skrip dari server
keyCheckEvent.OnClientEvent:Connect(function(scriptContent)
    if scriptContent then
        -- Menjalankan skrip yang diterima dari server
        local func, err = loadstring(scriptContent)
        if func then
            func()  -- Eksekusi skrip yang diterima
            print("Script executed successfully!")
        else
            print("Error executing script: " .. err)
        end
    end
end)
